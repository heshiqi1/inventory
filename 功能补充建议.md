# indicators/ema_group.py
class EMAGroup:
    def __init__(self, periods: list):
        self.periods = periods
    
    def calculate(self, df: pd.DataFrame) -> pd.DataFrame:
        """计算EMA均线组"""
        for period in self.periods:
            df[f'ema_{period}'] = df['Close'].ewm(span=period).mean()
        return df
    
    def is_bullish_arrangement(self, df: pd.DataFrame, idx: int) -> bool:
        """检查是否多头排列"""
        if idx < max(self.periods):
            return False
        emas = [df[f'ema_{p}'].iloc[idx] for p in self.periods]
        return all(emas[i] > emas[i+1] for i in range(len(emas)-1))
```

# filters/factor_detector.py
class FactorDetector:
    def __init__(self):
        self.unfavorable_detector = UnfavorableFactorDetector()
        self.favorable_detector = FavorableFactorDetector()
    
    def analyze(self, signal: dict, df: pd.DataFrame, idx: int) -> dict:
        """综合分析不利因素和有利因素"""
        unfavorable_score = self.unfavorable_detector.calculate_score(df, idx, signal)
        favorable_score = self.favorable_detector.calculate_score(df, idx, signal)
        
        return {
            'unfavorable_score': unfavorable_score,
            'favorable_score': favorable_score,
            'recommendation': self._get_recommendation(unfavorable_score, favorable_score)
        }
    
    def _get_recommendation(self, unfav_score: float, fav_score: float) -> str:
        """根据评分给出建议"""
        if unfav_score > 0.7:
            return 'light_position_or_skip'  # 轻仓或不参与
        elif fav_score > 0.7:
            return 'normal_position'  # 正常仓位
        else:
            return 'normal_position'  # 正常仓位
```
