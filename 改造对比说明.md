### 2️⃣ 消息发送函数 (第 51-113 行)

**Telegram 版本：**
```python
def send_telegram(message: str, silent: bool = False) -> bool:
    """发送 Telegram 消息"""
    if TELEGRAM_BOT_TOKEN == "YOUR_BOT_TOKEN":
        print(f"  [Telegram 模拟] {message[:80]}...")
        return True

    url = f"https://api.telegram.org/bot{TELEGRAM_BOT_TOKEN}/sendMessage"
    payload = {
        "chat_id": TELEGRAM_CHAT_ID,
        "text": message,
        "parse_mode": "HTML",
        "disable_notification": silent,
    }
    try:
        resp = requests.post(url, json=payload, timeout=10)
        resp.raise_for_status()
        return True
    except Exception as e:
        print(f"  [Telegram 发送失败] {e}")
        return False
```

**飞书版本：**
```python
def send_feishu(card_content: dict, silent: bool = False) -> bool:
    """发送飞书消息（交互式卡片）"""
    if FEISHU_WEBHOOK_URL == "YOUR_WEBHOOK_URL":
        print(f"  [飞书模拟] 卡片标题: {card_content.get('header', {}).get('title', {}).get('content', 'N/A')}")
        return True

    payload = {
        "msg_type": "interactive",
        "card": card_content
    }
    
    try:
        resp = requests.post(
            FEISHU_WEBHOOK_URL, 
            json=payload, 
            headers={"Content-Type": "application/json"},
            timeout=10
        )
        resp.raise_for_status()
        result = resp.json()
        if result.get('code') != 0:
            print(f"  [飞书发送失败] {result.get('msg', 'Unknown error')}")
            return False
        return True
    except Exception as e:
        print(f"  [飞书发送失败] {e}")
        return False


def send_feishu_text(message: str) -> bool:
    """发送飞书文本消息（用于简单通知）"""
    if FEISHU_WEBHOOK_URL == "YOUR_WEBHOOK_URL":
        print(f"  [飞书模拟文本] {message[:80]}...")
        return True

    payload = {
        "msg_type": "text",
        "content": {
            "text": message
        }
    }
    
    try:
        resp = requests.post(
            FEISHU_WEBHOOK_URL, 
            json=payload, 
            headers={"Content-Type": "application/json"},
            timeout=10
        )
        resp.raise_for_status()
        result = resp.json()
        return result.get('code') == 0
    except Exception as e:
        print(f"  [飞书发送失败] {e}")
        return False
```

✅ **改进：**
- 支持交互式卡片（美观）
- 支持纯文本消息（简单通知）
- 错误处理更完善（检查返回 code）
